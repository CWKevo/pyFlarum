468    def included(self) -> t.List[dict]:
469        """
470            Returns raw list of JSON included data.

473        """
474
475        return self.get("included", [{}])

478    def get_author(self, mode: t.Union[t.Any, 't.Literal["first_number"]']='first_number') -> UserFromBulk:
479        """
480            Obtains the discussion's author, AKA. the author
481            of the post with number 1 in a discussion.
482
483            `mode` allows you to specify the mode that is used to determine whether or not the post is the first post
484            of the discussion.
485
486            - `'first_number'` - checks if the number of the post is 1 - if yes, it fetches that post's author.
487            - `Any` - if anything other than `'first_number'` is passed (e. g. `'first_user``, but this can be anything), then this
488            returns the author of the first post in the JSON.
489            I am not sure how reliable is this, and whether or not the posts are actually ordered correctly in the API, so it's
490            probably a good idea to also check if the number of the post is 1 - but then again, what if the first post gets removed?
491        """
492
493        for post in self.get_posts():
494            if mode == 'first_number':
495                if post.number == 1:
496                    author = post.get_author()
497
498                    return author
499
500            else:
501                author = post.get_author()
502                return author
503
504        return None
505

507    def get_posts(self) -> t.List[PostFromBulk]:
508        """
509            Returns a list of `pyflarum.flarum.core.posts.PostFromBulk` objects.
510
511            It might seem strange why this doesn't return `pyflarum.flarum.core.posts.PostFromDiscussion` instead,
512            but these posts are in fact identical to `pyflarum.flarum.core.posts.PostFromBulk`, that's why they are returned.
513
514            `pyflarum.flarum.core.posts.PostFromDiscussion` comes from `pyflarum.flarum.core.discussions.DiscussionFromBulk` instead.
515        """
516
517        all_posts = list()
518        raw_posts = self.relationships.get("posts", {}).get("data", [{}]) # type: t.List[dict]
519
520        for raw_post in raw_posts:
521            if raw_post.get("type", None) == 'posts':
522                id_to_find = raw_post.get("id", None)
523
524                if id_to_find:
525                    for possible_raw_post in self.included:
526                        if (possible_raw_post.get("type", None) == 'posts') and (possible_raw_post.get("id", None) == id_to_find):
527                            post = PostFromBulk(user=self.user, _fetched_data=dict(data=possible_raw_post, _parent_included=self.included))
528
529                            all_posts.append(post)
530
531        return all_posts

mode allows you to specify the mode that is used to determine whether or not the post is the first post of the discussion.

  • 'first_number' - checks if the number of the post is 1 - if yes, it fetches that post's author.
  • Any - if anything other than 'first_number' is passed (e. g. `'first_user``, but this can be anything), then this returns the author of the first post in the JSON. I am not sure how reliable is this, and whether or not the posts are actually ordered correctly in the API, so it's probably a good idea to also check if the number of the post is 1 - but then again, what if the first post gets removed?
