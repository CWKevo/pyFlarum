 73    def authenticate(self, username_or_email: t.Optional[str]=None, password: t.Optional[str]=None) -> None:
 74        """
 75            Authenticates your user. This can be run after `FlarumUser` was initialized, to switch to a different user. You can even change
 76            `FlarumUser.forum_url` to login to another forum.
 77
 78            ### Parameters:
 79            - `username_or_email` - optional. The username or E-mail address to log into. If `None`, then the user isn't logged in.
 80            - `password` - optional. The user's password. If `None`, then the user isn't logged in.
 81        """
 82
 83        if username_or_email and password:
 84            identification_data = {"identification": username_or_email, "password": password}
 85
 86            raw = self.session.post(url=f'{self.forum_url}/api/token', json=identification_data)
 87            token_data = parse_request(raw)
 88
 89            token = token_data.get("token", None)
 90            user_id = token_data.get("userId", None)
 91
 92            if token:
 93                self.session.headers.update({
 94                    "Authorization": f'Token {token}; userId={user_id}',
 95                })
 96
 97                self.is_authenticated = True
 98
 99            else:
100                self.is_authenticated = False
101
102        else:
103            self.session.headers.update({
104                "Authorization": None,
105            })
106
107            self.is_authenticated = False

170    def _fetch_user_data(self) -> t.Optional[dict]:
171        """
172            Fetches your user's JSON data.
173        """
174
175        if self.username_or_email:
176            filter = Filter(limit=1)
177
178            if '@' in self.username_or_email:
179                filter.query = f'email:{self.username_or_email}'
180
181            else:
182                filter.query = self.username_or_email
183
184
185            raw = self.session.get(f"{self.api_urls['users']}", params=filter.to_dict)
186            json = parse_request(raw)
187
188            for possible_user in json.get("data", [{}]):
189                if '@' in self.username_or_email:
190                    if possible_user.get("attributes", {}).get("email", None) == self.username_or_email:
191                        return possible_user
192
193                else:
194                    if possible_user.get("attributes", {}).get("username", None) == self.username_or_email:
195                        return possible_user
196
197
198        return None

389        """
390            Updates the info of a user (this can be your user or someone else).

393            If you are updating someone else, then the updated `User` is returned.
394
395            ### Parameters:
396            - `user` - the user to update.
397            - `new_username` - the user's new username.
398            - `groups` - new groups of the user. This can either be a list of `pyflarum.flarum.core.groups.Group` objects,
399            or just one `pyflarum.flarum.core.groups.Groups`, or a list of `str` or `int` representing the group IDs.
400        """
401
402        id = user.id if user else self.data.id
403
404        patch_data = {
405            "data": {
406                "type": "users",
407                "id": id,
408                "attributes": {},
409                "relationships": {}
410            }
411        } # type: t.Dict[str, t.Dict[str, dict]]
412
413
414        if new_username:
415            patch_data['attributes']['username'] = new_username
416
417        if groups:
418            patch_data['data']['relationships'].update({"groups": {"data": [group.id if isinstance(group, Group) else group for group in groups]}})
419
420
421        raw = self.session.patch(f"{self.api_urls['users']}/{id}", json=patch_data)
422        json = parse_request(raw)
423
424
425        return self._update_user_data(new_data=json)
